cmake_minimum_required(VERSION 2.8.9)
project (neural_network)

option(debug "debug" OFF)
option(windows "windows" OFF)

if (windows)
	message("Building for Windows")
	#set(CMAKE_CXX_COMPILER g++.exe)
	set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11")
	#set(CMAKE_SHARED_LIBRARY_LINK_CXX_FLAGS "-static-libgcc -static-libstdc++ -static -lmingw32")
	#set (CMAKE_SHARED_LINKER_FLAGS "${CMAKE_SHARED_LINKER_FLAGS} -lmingw32 ")
	
	#set(CMAKE_CXX_FLAGS  "${CMAKE_CXX_FLAGS} -lmingw32 ")
	#set(CMAKE_EXE_LINKER_FLAGS  "${CMAKE_EXE_LINKER_FLAGS} -lmingw32 ")
	
	set(SDL2_PATH          "${PROJECT_SOURCE_DIR}/../../i686-w64-mingw32")
	set(SDL2_INCLUDE_DIR   "${PROJECT_SOURCE_DIR}/../../i686-w64-mingw32/include")
	set(SDL2_LIBRARY       "${PROJECT_SOURCE_DIR}/../../i686-w64-mingw32/lib/libSDL2.dll.a")
	set(SDL2_IMAGE_LIBRARY "${PROJECT_SOURCE_DIR}/../../i686-w64-mingw32/lib/libSDL2_image.dll.a")
	set(SDL2_TTF_LIBRARY   "${PROJECT_SOURCE_DIR}/../../i686-w64-mingw32/lib/libSDL2_ttf.dll.a")
	set(SDL2_MAIN          "${PROJECT_SOURCE_DIR}/../../i686-w64-mingw32/lib/libSDL2main.a")
else()
	message("Building for Linux")
	set(CMAKE_CXX_COMPILER /usr/bin/clang++)
	set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++14 -Wall -Wextra")
endif()

if (debug)
	set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g")
else()
	set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3")
endif()

set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${PROJECT_SOURCE_DIR}/../cmake")

find_package(SDL2 REQUIRED)
find_package(SDL2_ttf REQUIRED)
find_package(SDL2_image REQUIRED)
include_directories(${SDL2_INCLUDE_DIR} ${SDL2_TTF_INCLUDE_DIR} ${SDL2_IMAGE_INCLUDE_DIR})

file(GLOB_RECURSE SOURCES ../src/*.cpp ../src/*.hpp)

if (windows)
	add_executable(NeuralNetwork ${SOURCES})
	target_link_libraries(NeuralNetwork -lmingw32 ${SDL2_MAIN} ${SDL2_LIBRARY} ${SDL2_TTF_LIBRARY} ${SDL2_IMAGE_LIBRARY} )
else()
	add_executable(neural_network ${SOURCES})
	target_link_libraries(neural_network ${SDL2_LIBRARY} ${SDL2_TTF_LIBRARY} ${SDL2_IMAGE_LIBRARY})
endif()
